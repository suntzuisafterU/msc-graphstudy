307  template <typename FwdIterT, typename ValueT = typename std::iterator_traits<FwdIterT>::value_type>
308  std::enable_if_t<std::is_floating_point_v<ValueT>, stochastic_summary>
309  get_stochastic_summary(const FwdIterT first, const FwdIterT last)
310  {
311  auto sum = ValueT{};
312  auto sqrsum = ValueT{};
313  auto min = std::numeric_limits<ValueT>::max();
314  auto max = std::numeric_limits<ValueT>::min();
315  auto count = std::size_t{};
316  for (auto it = first; it != last; ++it) {
317  const auto value = *it;
318  sum += value;
319  sqrsum += square(value);
320  min = std::min(min, value);
321  max = std::max(max, value);
322  count += 1;
323  }
324  return {count, min, max, sum / count, std::sqrt(sqrsum / count)};
325  }

374  template <typename FwdIterT, typename ValueT = typename std::iterator_traits<FwdIterT>::value_type>
375  std::enable_if_t<std::is_floating_point_v<ValueT>, ValueT>
376  entropy(const FwdIterT first, const FwdIterT last)
377  {
378  constexpr auto zero = ValueT{0};
379  assert(std::all_of(first, last, [](const auto x){ return (x >= zero); }));
380  if (first == last) {
381  return zero;
382  }
383  assert(std::abs(1.0 - std::accumulate(first, last, zero)) < 1.0e-6);
384  const auto lambda = [](const auto acc, const auto x){
385  return (x > zero) ? acc + x * std::log2(x) : acc;
386  };
387  // We use `std::abs` rather than negation because we never want -0 as a result.
388  return std::abs(std::accumulate(first, last, zero, lambda));
389  }
